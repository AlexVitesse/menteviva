import os
from groq import Groq
from app.config import settings

client = Groq(api_key=settings.groq_api_key)

async def transcribe_audio(audio_bytes: bytes, filename: str = "audio.webm") -> str:
    """
    Transcribe audio usando Groq Whisper.

    Args:
        audio_bytes: Audio en bytes (webm, mp3, wav, etc.)
        filename: Nombre del archivo con extensión

    Returns:
        Texto transcrito
    """
    transcription = client.audio.transcriptions.create(
        file=(filename, audio_bytes),
        model=settings.groq_model_whisper,
        language="es",
        response_format="text"
    )
    return transcription

from typing import AsyncGenerator
from groq import Groq
from app.config import settings

client = Groq(api_key=settings.groq_api_key)

async def chat_stream(
    messages: list[dict],
    system_prompt: str
) -> AsyncGenerator[str, None]:
    """
    Genera respuesta del LLM en streaming.

    Args:
        messages: Historial de conversación [{"role": "user", "content": "..."}]
        system_prompt: Prompt del sistema con personalidad del avatar

    Yields:
        Tokens de la respuesta uno por uno
    """
    full_messages = [
        {"role": "system", "content": system_prompt},
        *messages
    ]

    stream = client.chat.completions.create(
        model=settings.groq_model_llm,
        messages=full_messages,
        temperature=0.7,
        max_tokens=500,
        stream=True
    )

    for chunk in stream:
        if chunk.choices[0].delta.content:
            yield chunk.choices[0].delta.content

import edge_tts
import asyncio
from typing import AsyncGenerator
from app.config import settings

# Voces disponibles en español México:
# - es-MX-JorgeNeural (masculina)
# - es-MX-DaliaNeural (femenina)
# - es-MX-BeatrizNeural (femenina)
# - es-MX-CandelaNeural (femenina)
# - es-MX-CecilioNeural (masculina)

AVATAR_VOICES = {
    "roberto": "es-MX-JorgeNeural",      # Director TI - voz masculina seria
    "maria": "es-MX-DaliaNeural",         # Gerente ventas - voz femenina
    "carlos": "es-MX-CecilioNeural",      # Proveedor - voz masculina
}

async def text_to_speech(
    text: str,
    avatar_id: str = "roberto"
) -> bytes:
    """
    Convierte texto a audio usando Edge TTS.

    Args:
        text: Texto a convertir
        avatar_id: ID del avatar para seleccionar voz

    Returns:
        Audio en bytes (MP3)
    """
    voice = AVATAR_VOICES.get(avatar_id, settings.tts_voice)

    communicate = edge_tts.Communicate(text, voice)
    audio_bytes = b""

    async for chunk in communicate.stream():
        if chunk["type"] == "audio":
            audio_bytes += chunk["data"]

    return audio_bytes


async def text_to_speech_stream(
    text: str,
    avatar_id: str = "roberto"
) -> AsyncGenerator[bytes, None]:
    """
    Stream de audio para respuestas más largas.
    """
    voice = AVATAR_VOICES.get(avatar_id, settings.tts_voice)
    communicate = edge_tts.Communicate(text, voice)

    async for chunk in communicate.stream():
        if chunk["type"] == "audio":
            yield chunk["data"]

def get_avatar(avatar_id: str) -> dict:
    return AVATARS.get(avatar_id)

def get_all_avatars() -> list[dict]:
    return [
        {k: v for k, v in avatar.items() if k != "system_prompt"}
        for avatar in AVATARS.values()
    ]

def get_system_prompt(avatar_id: str) -> str:
    avatar = AVATARS.get(avatar_id)
    return avatar["system_prompt"] if avatar else ""

@router.get("/avatars")
async def list_avatars():
    """Lista todos los avatares disponibles."""

@router.get("/avatars/{avatar_id}")
async def get_avatar_detail(avatar_id: str):
    """Detalle de un avatar específico."""
    avatar = get_avatar(avatar_id)
    if not avatar:
        return {"error": "Avatar not found"}, 404
    # No exponer system_prompt
    return {k: v for k, v in avatar.items() if k != "system_prompt"}

import json
import base64
from fastapi import APIRouter, WebSocket, WebSocketDisconnect
from app.services.groq_whisper import transcribe_audio
from app.services.groq_llm import chat_stream
from app.services.edge_tts import text_to_speech
from app.prompts.scenarios import get_system_prompt

router = APIRouter()

@router.websocket("/conversation/{avatar_id}")
async def conversation_websocket(websocket: WebSocket, avatar_id: str):
    """
    WebSocket para conversación en tiempo real.

    Flujo:
    1. Cliente envía audio (base64)
    2. Server transcribe con Whisper
    3. Server genera respuesta con LLM (streaming)
    4. Server genera audio con TTS
    5. Server envía texto + audio al cliente
    """
    await websocket.accept()

    system_prompt = get_system_prompt(avatar_id)
    if not system_prompt:
        await websocket.send_json({"error": "Avatar not found"})
        await websocket.close()
        return

    conversation_history = []

    try:
        while True:
            # Recibir mensaje del cliente
            data = await websocket.receive_json()

            if data.get("type") == "audio":
                # 1. Decodificar audio
                audio_base64 = data.get("audio")
                audio_bytes = base64.b64decode(audio_base64)

                # 2. Transcribir con Whisper
                await websocket.send_json({
                    "type": "status",
                    "status": "transcribing"
                })

                user_text = await transcribe_audio(audio_bytes)

                await websocket.send_json({
                    "type": "user_message",
                    "content": user_text
                })

                # 3. Agregar al historial
                conversation_history.append({
                    "role": "user",
                    "content": user_text
                })

                # 4. Generar respuesta con LLM (streaming)
                await websocket.send_json({
                    "type": "status",
                    "status": "thinking"
                })

                full_response = ""
                async for token in chat_stream(conversation_history, system_prompt):
                    full_response += token
                    await websocket.send_json({
                        "type": "assistant_token",
                        "content": token
                    })

                # 5. Agregar respuesta al historial
                conversation_history.append({
                    "role": "assistant",
                    "content": full_response
                })

                # 6. Generar audio con TTS
                await websocket.send_json({
                    "type": "status",
                    "status": "generating_audio"
                })

                audio_response = await text_to_speech(full_response, avatar_id)
                audio_base64_response = base64.b64encode(audio_response).decode()

                await websocket.send_json({
                    "type": "assistant_audio",
                    "audio": audio_base64_response,
                    "content": full_response
                })

                await websocket.send_json({
                    "type": "status",
                    "status": "ready"
                })

            elif data.get("type") == "end_session":
                # Calcular métricas básicas
                total_exchanges = len(conversation_history) // 2
                await websocket.send_json({
                    "type": "session_end",
                    "metrics": {
                        "total_exchanges": total_exchanges,
                        "conversation": conversation_history
                    }
                })
                break

    except WebSocketDisconnect:
        print(f"Cliente desconectado de conversación con {avatar_id}")
